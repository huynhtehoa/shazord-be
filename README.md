```bash
  # install venv
  python3 -m venv .venv

  # install ffmpeg
  sudo apt install ffmpeg

  # create env
  touch .env

  # activate venv
  source .venv/bin/activate

  # install dependencies
  pip install -r requirements.txt

  # start server
  ./start_dev

  # deactivate venv
  deactivate
```
